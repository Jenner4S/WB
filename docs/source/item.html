/**
 * @class General.item
 * 
 * 工具条、菜单或其他容器的子项控件，该控件可以根据上级容器控件自动生成对应的子控件。
 * 例如在工具条下添加默认生成工具按钮，在菜单下添加默认生成菜单项，在该控件下添加子控件将生成子菜单等。
 * 该控件封装了Ext.toolbar.Item, Ext.menu.Item和Ext.menu.Menu等多个类似功能的控件，扩展属性在本文档中说明，其他属性请参考Ext JS文档说明。
 */
var item = {
  /**
   * @cfg {String} bindModule
   * 当前用户如果不能访问该属性指定的模块，则该控件隐藏。
   */

  /**
   * @cfg {Boolean} createInstance
   * 是否创建对象的实例，如果设置为false只创建对象表达式。
   */
  createInstance: true,

  /**
   * @cfg {Boolean} ignoreEvents
   * 是否取消默认的菜单项下述事件：键导航, onMouseOver和onMouseLeave。设置该属性为true可用于在菜单中添加编辑控件，以使编辑控件使用正常。
   */
  ignoreEvents: false,

  /**
   * @cfg {Boolean} isConfig
   * 指定该控件是否作为父控件的配置项。true作为父控件的配置项，false作为父控件的子控件。
   * 如果值为true，该控件的对象表达式将添加到父控件的以该控件itemId属性值为名称的属性下。
   */
  isConfig: 'auto',

  /**
   * @cfg {String} menuType
   * 当控件为菜单时指定菜单的xtype类型，可选项为colormenu和datemenu。
   */

  /**
   * @cfg {String} normalName
   * 在命名空间下指向该对象的变量名称，可以通过app.normalName访问该对象。
   * 如果该值为空，normalName取自itemId属性值。
   */
  normalName: undefined,

  /**
   * @cfg {Object} tagConfigs
   * 对象或任意的脚本添加到该控件的配置项中。如设置为{foo: 'bar'}将为该控件增加配置项foo，其值为bar。
   */
  tagConfigs: undefined

  /**
   * @cfg {Object} toolcallback
   * 当控件为面板标题中的工具条按钮时，点击按钮执行的方法。见Ext.panel.Tool的callback属性。
   */

  /**
   * @cfg {Object} tooltype
   * 当控件为面板标题中的工具条按钮时，指定按钮的类型。见Ext.panel.Tool的type属性。
   */

  /**
   * @event menubeforeshow
   * 当控件为菜单时菜单显示之前触发的事件。见Ext.menu.Menu的beforeshow事件。
   */

  /**
   * @event menuclick
   * 当控件为菜单时菜单项的点击事件。见Ext.menu.Menu的click事件。
   */

  /**
   * @event menuonshow
   * 当控件为菜单时菜单的显示事件。见Ext.menu.Menu的show事件。
   */

  /**
   * @event menuselect
   * 当控件为日期或颜色菜单时的菜单选择事件。见Ext.menu.DatePicker/Ext.menu.ColorPicker的select事件。
   */

  /**
   * @event tagEvents
   * 对象或任意的脚本添加到该控件的事件中。如设置为{destroy: app.destroyHandler}将为该控件增加destroy事件时执行app.destroyHandler方法。
   */
};
